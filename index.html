Sample Analytical SQL 
-- Occupancy rate by building
SELECT b.Building_ID,
       ROUND(SUM(CASE WHEN CURDATE() BETWEEN l.Start_Date AND l.End_Date THEN 1 ELSE 0 END)
             / NULLIF(COUNT(DISTINCT a.Apartment_ID),0), 3) AS occupancy_rate
FROM BUILDING b
JOIN APARTMENT a ON a.Building_ID = b.Building_ID
LEFT JOIN LEASE l ON l.Apartment_ID = a.Apartment_ID
                 AND l.Building_ID = a.Building_ID
GROUP BY b.Building_ID;

-- Average maintenance request turnaround (days)
SELECT Request_Type,
       ROUND(AVG(DATEDIFF(Completion_Date, Request_Date)), 2) AS avg_days
FROM MAINTENANCE_REQUEST
WHERE Completion_Date IS NOT NULL
GROUP BY Request_Type;
